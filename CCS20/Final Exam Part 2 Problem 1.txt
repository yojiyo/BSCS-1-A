package finalexampart2;
import java.util.Scanner;
public class Problem1 {

	public static void main(String[] args) {

		String top = "top";
		String bottom = "bottom";
		String right = "right";
		String left = "left";
		
		Scanner in = new Scanner(System.in);
		System.out.println("Input a number.");
			int n = in.nextInt();
			
		Scanner in2 = new Scanner(System.in);			
		System.out.println("Select starting point: top or bottom");
			String option1 = in2.next();		
		
		Scanner in3 = new Scanner(System.in);
		System.out.println("Select ending point: right or left");
			String option2 = in3.next();	
		
		if (option1.equalsIgnoreCase(top)) {
			for (int i = 1; i <= n; i++) {
				if (option2.equalsIgnoreCase(right)) {
					for (int j = 1; j <= i; j++) {
						System.out.print("*  ");
					}	
				}
				else if (option2.equalsIgnoreCase(left)) {
					for (int j = 1; j <= (-1 * i + n); j++) {
						System.out.print("   ");
					}
					for (int k = 1; k <= i; k++) {
						System.out.print("*  ");
					}
				}
				System.out.println();	
			}
		}
		else if (option1.equalsIgnoreCase(bottom)) {
			for (int i = n; i >= 1; i--) {
				if (option2.equalsIgnoreCase(right)) {
					for (int j = i; j >= 1; j--) {
						System.out.print("*  ");
					}	
				}
				else if (option2.equalsIgnoreCase(left)) {
					for (int j = 1; j <= (-1 * i + n); j++) {
						System.out.print("   ");
					}
					for (int k = 1; k <= i; k++) {
						System.out.print("*  ");
					}
				}
				System.out.println();	
			}
		}
	}
}
